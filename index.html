        4. Write a code to calculate area of a rectangle using inner function. In this case outer function should accept parameter length and inner function 
        should accept parameter breadth.
    
        O/P: 
        //Using currying
        function calculateArea(length) {
            return function (breadth) {
                return length * breadth;
            }
        }
        console.log(calculateArea(4)(5));

        //using self invocation function inside
        function area(length) {
            (function bred(breadth) {
                console.log(length * breadth);
            })(20);
        }

        area(10);

        //Using normal functionality
        function area(length) {
            function bred(breadth) {
                return length * breadth;
            };

            return bred(20);
        }

        console.log(area(10));
